# This is used to stretch and rotate 2d graph, 
# x,y is how much and also the direction you want to stretch. 
# Example: x = -1 means flip alongside x axis. x=2 means stretch twice alongside x axis.
# alpha is the angle you want to rotate.
# df is the dataset.
def transform_rotate(x, y, alpha, df):
    a = np.array([df.x, df.y])
    Fx = np.array([[x, 0],[0, y]])
    a_fliped = np.dot(Fx, a)
    
    R = np.array([[np.cos(-alpha*np.pi/180), np.sin(-alpha*np.pi/180)], 
              [-np.sin(-alpha*np.pi/180), np.cos(-alpha*np.pi/180)]])
    
    temp1 = np.dot(np.dot(R.T, Fx), a)
    print('The corresponding transition matrix is: ')
    print(np.dot(R.T, Fx))
    
    plt.plot(temp1[0], temp1[1])
    plt.axis('equal')

## In this part I made a function to do the stretch and rotation at the
## same time.
## change xx1, yy2 to manipulate the A = I matrix. 
def f(x, xx1, yy1):
    A = np.array([[xx1,0],[0,yy1]])
    return np.dot(x.T, np.dot(A,x))

def plotContour(xx1,yy1): 
    A = np.array([[xx1,0],[0,yy1]])
    print('A:')
    print(A)
    nGrid = 30 
    x1 = np.linspace(-4,4,nGrid)
    x2 = np.linspace(-4,4,nGrid)
    y = np.empty((nGrid, nGrid))
    for i in range(len(x1)):
        for j in range(len(x2)):
            y[i,j] = f(np.array([x1[i], x2[j]]),xx1,yy1)

    fig = plt.figure()
    ax = fig.add_subplot(111)
    a = ax.contour(x1, x2, y)
    ax.clabel(a, inline=1, fontsize=8)
    # ax.set_title('Simplest default with labels')
    ax.set_aspect('equal')
    plt.show()
